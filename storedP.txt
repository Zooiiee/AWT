table
CREATE TABLE Users
(
    Id INT PRIMARY KEY,
    Name NVARCHAR(100),
    Password NVARCHAR(100),
    Email NVARCHAR(100)
);

procedure
CREATE PROCEDURE InsertUser
    @Id INT,
    @Name NVARCHAR(100),
    @Password NVARCHAR(100),
    @Email NVARCHAR(100)
AS
BEGIN
    INSERT INTO Users (Id, Name, Password, Email)
    VALUES (@Id, @Name, @Password, @Email)
END
